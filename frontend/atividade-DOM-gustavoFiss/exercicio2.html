Cores Alternadas

Índice ímpar → vermelho, índice par → azul. Usa style.color dentro de um loop.

function alternarCores() { const paragrafos = document.getElementsByTagName('p'); for (let i = 0; i < paragrafos.length; i++) { // índice ímpar (1, 3, 5...) → vermelho // índice par (0, 2, 4...) → azul if (i % 2 !== 0) { paragrafos[i].style.color = 'red'; } else { paragrafos[i].style.color = 'blue'; } } }
